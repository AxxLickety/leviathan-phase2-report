import numpy as np
import pandas as pd
import matplotlib.pyplot as plt

pd.set_option("display.width", 140)
pd.set_option("display.max_columns", 200)

def sharpe(x):
    x = pd.Series(x).dropna()
    s = x.std(ddof=1)
    return float(x.mean() / s) if s > 0 else np.nan

def summarize(x):
    x = pd.Series(x).dropna()
    m = x.mean()
    v = x.std(ddof=1)
    return {
        "n": int(len(x)),
        "mean": float(m),
        "vol": float(v),
        "sharpe": float(m / v) if v > 0 else np.nan,
        "p05": float(x.quantile(0.05)),
        "p10": float(x.quantile(0.10)),
        "min": float(x.min()),
        "max": float(x.max()),
    }

def equity_from_logret(x):
    return np.exp(pd.Series(x).fillna(0).cumsum())

def drawdown(eq):
    eq = pd.Series(eq)
    peak = eq.cummax()
    return eq / peak - 1.0

def oos_walkforward_once(core_df, split_date, q_supply=0.90):
    full = core_df.copy()
    split = pd.Timestamp(split_date)

    train = full[full[col_date] <= split].copy()
    test  = full[full[col_date] >  split].copy()

    thr_supply = train[col_supply_val].quantile(q_supply)

    test["supply_high_oos"] = (test[col_supply_val] >= thr_supply)
    test["invested_oos"] = ~((test["regime"].eq(0)) & (test["supply_high_oos"]))
    test["strat_oos"] = test[col_ret].where(test["invested_oos"], 0.0)

    out = {
        "split": str(split.date()),
        "q_supply": q_supply,
        "thr_supply_train": float(thr_supply),
        "oos_n": int(len(test)),
        "oos_exposure": float(test["invested_oos"].mean()),
        "oos_baseline": summarize(test[col_ret]),
        "oos_strat": summarize(test["strat_oos"]),
    }
    return out, test

rng = np.random.default_rng(0)

def random_off_null(ret, off_count, n_sims=5000):
    ret = pd.Series(ret).reset_index(drop=True)
    T = len(ret)
    sharpes = []
    maxdds = []
    for _ in range(n_sims):
        off_idx = rng.choice(T, size=off_count, replace=False)
        strat = ret.copy()
        strat.iloc[off_idx] = 0.0
        sharpes.append(sharpe(strat))
        maxdds.append(float(drawdown(equity_from_logret(strat)).min()))
    return np.array(sharpes), np.array(maxdds)

def circular_shift_null(ret, off_mask, n_sims=5000):
    ret = pd.Series(ret).reset_index(drop=True)
    off_mask = pd.Series(off_mask).reset_index(drop=True).astype(bool)
    T = len(ret)

    sharpes = []
    maxdds = []
    for k in rng.integers(0, T, size=n_sims):
        shifted_off = off_mask.shift(k, fill_value=False)  # shift creates NaN; fill False then wrap manually
        # wrap manually using roll
        shifted_off = pd.Series(np.roll(off_mask.values, k))

        strat = ret.where(~shifted_off, 0.0)
        sharpes.append(sharpe(strat))
        maxdds.append(float(drawdown(equity_from_logret(strat)).min()))
    return np.array(sharpes), np.array(maxdds)

def equity_from_logret(x):
    return np.exp(pd.Series(x).fillna(0).cumsum())

def drawdown(eq):
    eq = pd.Series(eq)
    peak = eq.cummax()
    return eq / peak - 1.0
